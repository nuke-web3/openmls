76    /// Join an existing group through an External Commit.
77    /// The resulting [`MlsGroup`] instance starts off with a pending
78    /// commit (the external commit, which adds this client to the group).
79    /// Merging this commit is necessary for this [`MlsGroup`] instance to
80    /// function properly, as, for example, this client is not yet part of the
81    /// tree. As a result, it is not possible to clear the pending commit. If
82    /// the external commit was rejected due to an epoch change, the
83    /// [`MlsGroup`] instance has to be discarded and a new one has to be
84    /// created using this function based on the latest `ratchet_tree` and
85    /// group info. For more information on the external init process,
86    /// please see Section 11.2.1 in the MLS specification.
87    ///
88    /// Note: If there is a group member in the group with the same identity as
89    /// us, this will create a remove proposal.
90    #[allow(clippy::too_many_arguments)]
91    pub fn join_by_external_commit<Provider: OpenMlsProvider>(
92        provider: &Provider,
93        signer: &impl Signer,
94        ratchet_tree: Option<RatchetTreeIn>,
95        verifiable_group_info: VerifiableGroupInfo,
96        mls_group_config: &MlsGroupJoinConfig,
97        capabilities: Option<Capabilities>,
98        extensions: Option<Extensions>,
99        aad: &[u8],
100        credential_with_key: CredentialWithKey,
101    ) -> Result<(Self, MlsMessageOut, Option<GroupInfo>), ExternalCommitError<Provider::StorageError>>
102    {
103        // Prepare the commit parameters
104        let framing_parameters = FramingParameters::new(aad, WireFormat::PublicMessage);
105
106        let leaf_node_parameters = LeafNodeParameters::builder()
107            .with_capabilities(capabilities.unwrap_or_default())
108            .with_extensions(extensions.unwrap_or_default())
109            .build();
110        let mut params = CreateCommitParams::builder()
111            .external_commit(credential_with_key, framing_parameters)
112            .leaf_node_parameters(leaf_node_parameters)
113            .build();
114
115        // Build the ratchet tree
116
117        // Set nodes either from the extension or from the `nodes_option`.
118        // If we got a ratchet tree extension in the welcome, we enable it for
119        // this group. Note that this is not strictly necessary. But there's
120        // currently no other mechanism to enable the extension.
121        let ratchet_tree = match verifiable_group_info.extensions().ratchet_tree() {
122            Some(extension) => extension.ratchet_tree().clone(),
123            None => match ratchet_tree {
124                Some(ratchet_tree) => ratchet_tree,
125                None => return Err(ExternalCommitError::MissingRatchetTree),
126            },
127        };
128
129        let (public_group, group_info) = PublicGroup::from_external_internal(
130            provider.crypto(),
131            ratchet_tree,
132            verifiable_group_info,
133            // Existing proposals are discarded when joining by external commit.
134            ProposalStore::new(),
135        )?;
136        let group_context = public_group.group_context();
137
138        // Obtain external_pub from GroupInfo extensions.
139        let external_pub = group_info
140            .extensions()
141            .external_pub()
142            .ok_or(ExternalCommitError::MissingExternalPub)?
143            .external_pub();
144
145        let (init_secret, kem_output) = InitSecret::from_group_context(
146            provider.crypto(),
147            group_context,
148            external_pub.as_slice(),
149        )
150        .map_err(|_| ExternalCommitError::UnsupportedCiphersuite)?;
151
152        // The `EpochSecrets` we create here are essentially zero, with the
153        // exception of the `InitSecret`, which is all we need here for the
154        // external commit.
155        let ciphersuite = group_info.group_context().ciphersuite();
156        let epoch_secrets =
157            EpochSecrets::with_init_secret(provider.crypto(), ciphersuite, init_secret)
158                .map_err(LibraryError::unexpected_crypto_error)?;
159        let (group_epoch_secrets, message_secrets) = epoch_secrets.split_secrets(
160            group_context
161                .tls_serialize_detached()
162                .map_err(LibraryError::missing_bound_check)?,
163            public_group.tree_size(),
164            // We use a fake own index of 0 here, as we're not going to use the
165            // tree for encryption until after the first commit. This issue is
166            // tracked in #767.
167            LeafNodeIndex::new(0u32),
168        );
169        let message_secrets_store = MessageSecretsStore::new_with_secret(0, message_secrets);
170
171        let external_init_proposal = Proposal::ExternalInit(ExternalInitProposal::from(kem_output));
172
173        let mut inline_proposals = vec![external_init_proposal];
174
175        // If there is a group member in the group with the same identity as us,
176        // commit a remove proposal.
177        let signature_key = params.credential_with_key().signature_key.as_slice();
178        if let Some(us) = public_group
179            .members()
180            .find(|member| member.signature_key == signature_key)
181        {
182            let remove_proposal = Proposal::Remove(RemoveProposal { removed: us.index });
183            inline_proposals.push(remove_proposal);
184        };
185
186        let own_leaf_index =
187            public_group.leftmost_free_index(inline_proposals.iter(), iter::empty())?;
188        params.set_inline_proposals(inline_proposals);
189
190        let mut mls_group = MlsGroup {
191            mls_group_config: mls_group_config.clone(),
192            own_leaf_nodes: vec![],
193            aad: vec![],
194            group_state: MlsGroupState::Operational,
195            public_group,
196            group_epoch_secrets,
197            own_leaf_index,
198            message_secrets_store,
199            resumption_psk_store: ResumptionPskStore::new(32),
200        };
201
202        mls_group.set_max_past_epochs(mls_group_config.max_past_epochs);
203
204        // Immediately create the commit to add ourselves to the group.
205        let create_commit_result = mls_group
206            .create_external_commit(params, provider, signer)
207            .map_err(|_| ExternalCommitError::CommitError)?;
208
209        mls_group.group_state = MlsGroupState::PendingCommit(Box::new(
210            PendingCommitState::External(create_commit_result.staged_commit),
211        ));
212
213        mls_group
214            .store(provider.storage())
215            .map_err(ExternalCommitError::StorageError)?;
216
217        let public_message: PublicMessage = create_commit_result.commit.into();
218
219        Ok((
220            mls_group,
221            public_message.into(),
222            create_commit_result.group_info,
223        ))
224    }

227impl ProcessedWelcome {
228    /// Creates a new processed [`Welcome`] message , which can be
229    /// inspected before creating a [`StagedWelcome`].
230    ///
231    /// This does not require a ratchet tree yet.
232    ///
233    /// [`Welcome`]: crate::messages::Welcome
234    pub fn new_from_welcome<Provider: OpenMlsProvider>(
235        provider: &Provider,
236        mls_group_config: &MlsGroupJoinConfig,
237        welcome: Welcome,
238    ) -> Result<Self, WelcomeError<Provider::StorageError>> {
239        let (resumption_psk_store, key_package_bundle) =
240            keys_for_welcome(mls_group_config, &welcome, provider)?;
241
242        let ciphersuite = welcome.ciphersuite();
243        let Some(egs) = welcome.find_encrypted_group_secret(
244            key_package_bundle
245                .key_package()
246                .hash_ref(provider.crypto())?,
247        ) else {
248            return Err(WelcomeError::JoinerSecretNotFound);
249        };
250
251        // This check seems to be superfluous from the perspective of the RFC, but still doesn't
252        // seem like a bad idea.
253        if welcome.ciphersuite() != key_package_bundle.key_package().ciphersuite() {
254            let e = WelcomeError::CiphersuiteMismatch;
255            log::debug!("new_from_welcome {e:?}");
256            return Err(e);
257        }
258
259        let group_secrets = GroupSecrets::try_from_ciphertext(
260            key_package_bundle.init_private_key(),
261            egs.encrypted_group_secrets(),
262            welcome.encrypted_group_info(),
263            ciphersuite,
264            provider.crypto(),
265        )?;
266        let psk_secret = {
267            let psks = load_psks(
268                provider.storage(),
269                &resumption_psk_store,
270                &group_secrets.psks,
271            )?;
272
273            PskSecret::new(provider.crypto(), ciphersuite, psks)?
274        };
275        let key_schedule = KeySchedule::init(
276            ciphersuite,
277            provider.crypto(),
278            &group_secrets.joiner_secret,
279            psk_secret,
280        )?;
281        let (welcome_key, welcome_nonce) = key_schedule
282            .welcome(provider.crypto(), ciphersuite)
283            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?
284            .derive_welcome_key_nonce(provider.crypto(), ciphersuite)
285            .map_err(LibraryError::unexpected_crypto_error)?;
286        let verifiable_group_info = VerifiableGroupInfo::try_from_ciphertext(
287            &welcome_key,
288            &welcome_nonce,
289            welcome.encrypted_group_info(),
290            &[],
291            provider.crypto(),
292        )?;
293        if let Some(required_capabilities) =
294            verifiable_group_info.extensions().required_capabilities()
295        {
296            // Also check that our key package actually supports the extensions.
297            // As per the spec, the sender must have checked this. But you never know.
298            key_package_bundle
299                .key_package()
300                .leaf_node()
301                .capabilities()
302                .supports_required_capabilities(required_capabilities)?;
303        }
304
305        // https://validation.openmls.tech/#valn1404
306        // Verify that the cipher_suite in the GroupInfo matches the cipher_suite in the
307        // KeyPackage.
308        if verifiable_group_info.ciphersuite() != key_package_bundle.key_package().ciphersuite() {
309            let e = WelcomeError::CiphersuiteMismatch;
310            log::debug!("new_from_welcome {e:?}");
311            return Err(e);
312        }
313
314        Ok(Self {
315            mls_group_config: mls_group_config.clone(),
316            ciphersuite,
317            group_secrets,
318            key_schedule,
319            verifiable_group_info,
320            resumption_psk_store,
321            key_package_bundle,
322        })
323    }

339    /// Consume the `ProcessedWelcome` and combine it witht he ratchet tree into
340    /// a `StagedWelcome`.
341    pub fn into_staged_welcome<Provider: OpenMlsProvider>(
342        mut self,
343        provider: &Provider,
344        ratchet_tree: Option<RatchetTreeIn>,
345    ) -> Result<StagedWelcome, WelcomeError<Provider::StorageError>> {
346        // Build the ratchet tree and group
347
348        // Set nodes either from the extension or from the `nodes_option`.
349        // If we got a ratchet tree extension in the welcome, we enable it for
350        // this group. Note that this is not strictly necessary. But there's
351        // currently no other mechanism to enable the extension.
352        let ratchet_tree = match self.verifiable_group_info.extensions().ratchet_tree() {
353            Some(extension) => extension.ratchet_tree().clone(),
354            None => match ratchet_tree {
355                Some(ratchet_tree) => ratchet_tree,
356                None => return Err(WelcomeError::MissingRatchetTree),
357            },
358        };
359
360        // Since there is currently only the external pub extension, there is no
361        // group info extension of interest here.
362        let (public_group, _group_info_extensions) = PublicGroup::from_external_internal(
363            provider.crypto(),
364            ratchet_tree,
365            self.verifiable_group_info.clone(),
366            ProposalStore::new(),
367        )?;
368
369        // Find our own leaf in the tree.
370        let own_leaf_index = public_group
371            .members()
372            .find_map(|m| {
373                if m.signature_key
374                    == self
375                        .key_package_bundle
376                        .key_package()
377                        .leaf_node()
378                        .signature_key()
379                        .as_slice()
380                {
381                    Some(m.index)
382                } else {
383                    None
384                }
385            })
386            .ok_or(WelcomeError::PublicTreeError(
387                PublicTreeError::MalformedTree,
388            ))?;
389
390        let (group_epoch_secrets, message_secrets) = {
391            let serialized_group_context = public_group
392                .group_context()
393                .tls_serialize_detached()
394                .map_err(LibraryError::missing_bound_check)?;
395
396            // TODO #751: Implement PSK
397            self.key_schedule
398                .add_context(provider.crypto(), &serialized_group_context)
399                .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
400
401            let epoch_secrets = self
402                .key_schedule
403                .epoch_secrets(provider.crypto(), self.ciphersuite)
404                .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
405
406            epoch_secrets.split_secrets(
407                serialized_group_context,
408                public_group.tree_size(),
409                own_leaf_index,
410            )
411        };
412
413        let confirmation_tag = message_secrets
414            .confirmation_key()
415            .tag(
416                provider.crypto(),
417                self.ciphersuite,
418                public_group.group_context().confirmed_transcript_hash(),
419            )
420            .map_err(LibraryError::unexpected_crypto_error)?;
421
422        // Verify confirmation tag
423        // https://validation.openmls.tech/#valn1410
424        if &confirmation_tag != public_group.confirmation_tag() {
425            log::error!("Confirmation tag mismatch");
426            log_crypto!(trace, "  Got:      {:x?}", confirmation_tag);
427            log_crypto!(trace, "  Expected: {:x?}", public_group.confirmation_tag());
428            debug_assert!(false, "Confirmation tag mismatch");
429
430            // in some tests we need to be able to proceed despite the tag being wrong,
431            // e.g. to test whether a later validation check is performed correctly.
432            if !crate::skip_validation::is_disabled::confirmation_tag() {
433                return Err(WelcomeError::ConfirmationTagMismatch);
434            }
435        }
436
437        let message_secrets_store = MessageSecretsStore::new_with_secret(0, message_secrets);
438
439        // Extract and store the resumption PSK for the current epoch.
440        let resumption_psk = group_epoch_secrets.resumption_psk();
441        self.resumption_psk_store
442            .add(public_group.group_context().epoch(), resumption_psk.clone());
443
444        let welcome_sender_index = self.verifiable_group_info.signer();
445        let path_keypairs = if let Some(path_secret) = self.group_secrets.path_secret {
446            let (path_keypairs, _commit_secret) = public_group
447                .derive_path_secrets(
448                    provider.crypto(),
449                    self.ciphersuite,
450                    path_secret,
451                    welcome_sender_index,
452                    own_leaf_index,
453                )
454                .map_err(|e| match e {
455                    DerivePathError::LibraryError(e) => e.into(),
456                    DerivePathError::PublicKeyMismatch => {
457                        WelcomeError::PublicTreeError(PublicTreeError::PublicKeyMismatch)
458                    }
459                })?;
460            Some(path_keypairs)
461        } else {
462            None
463        };
464
465        let staged_welcome = StagedWelcome {
466            mls_group_config: self.mls_group_config,
467            public_group,
468            group_epoch_secrets,
469            own_leaf_index,
470            message_secrets_store,
471            resumption_psk_store: self.resumption_psk_store,
472            verifiable_group_info: self.verifiable_group_info,
473            key_package_bundle: self.key_package_bundle,
474            path_keypairs,
475        };
476
477        Ok(staged_welcome)
478    }

481impl StagedWelcome {
482    /// Creates a new staged welcome from a [`Welcome`] message. Returns an error
483    /// ([`WelcomeError::NoMatchingKeyPackage`]) if no [`KeyPackage`]
484    /// can be found.
485    /// Note: calling this function will consume the key material for decrypting the [`Welcome`]
486    /// message, even if the caller does not turn the [`StagedWelcome`] into an [`MlsGroup`].
487    ///
488    /// [`Welcome`]: crate::messages::Welcome
489    pub fn new_from_welcome<Provider: OpenMlsProvider>(
490        provider: &Provider,
491        mls_group_config: &MlsGroupJoinConfig,
492        welcome: Welcome,
493        ratchet_tree: Option<RatchetTreeIn>,
494    ) -> Result<Self, WelcomeError<Provider::StorageError>> {
495        let processed_welcome =
496            ProcessedWelcome::new_from_welcome(provider, mls_group_config, welcome)?;
497
498        processed_welcome.into_staged_welcome(provider, ratchet_tree)
499    }

530    /// Consumes the [`StagedWelcome`] and returns the respective [`MlsGroup`].
531    pub fn into_group<Provider: OpenMlsProvider>(
532        self,
533        provider: &Provider,
534    ) -> Result<MlsGroup, WelcomeError<Provider::StorageError>> {
535        // If we got a path secret, derive the path (which also checks if the
536        // public keys match) and store the derived keys in the key store.
537        let group_keypairs = if let Some(path_keypairs) = self.path_keypairs {
538            let mut keypairs = vec![self.key_package_bundle.encryption_key_pair()];
539            keypairs.extend_from_slice(&path_keypairs);
540            keypairs
541        } else {
542            vec![self.key_package_bundle.encryption_key_pair()]
543        };
544
545        let mut mls_group = MlsGroup {
546            mls_group_config: self.mls_group_config,
547            own_leaf_nodes: vec![],
548            aad: vec![],
549            group_state: MlsGroupState::Operational,
550            public_group: self.public_group,
551            group_epoch_secrets: self.group_epoch_secrets,
552            own_leaf_index: self.own_leaf_index,
553            message_secrets_store: self.message_secrets_store,
554            resumption_psk_store: self.resumption_psk_store,
555        };
556
557        mls_group
558            .store_epoch_keypairs(provider.storage(), group_keypairs.as_slice())
559            .map_err(WelcomeError::StorageError)?;
560        mls_group.set_max_past_epochs(mls_group.mls_group_config.max_past_epochs);
561
562        mls_group
563            .store(provider.storage())
564            .map_err(WelcomeError::StorageError)?;
565
566        Ok(mls_group)
567    }

570fn keys_for_welcome<Provider: OpenMlsProvider>(
571    mls_group_config: &MlsGroupJoinConfig,
572    welcome: &Welcome,
573    provider: &Provider,
574) -> Result<
575    (ResumptionPskStore, KeyPackageBundle),
576    WelcomeError<<Provider as OpenMlsProvider>::StorageError>,
577> {
578    let resumption_psk_store = ResumptionPskStore::new(mls_group_config.number_of_resumption_psks);
579    let key_package_bundle: KeyPackageBundle = welcome
580        .secrets()
581        .iter()
582        .find_map(|egs| {
583            let hash_ref = egs.new_member();
584
585            provider
586                .storage()
587                .key_package(&hash_ref)
588                .map_err(WelcomeError::StorageError)
589                .transpose()
590        })
591        .ok_or(WelcomeError::NoMatchingKeyPackage)??;
592    if !key_package_bundle.key_package().last_resort() {
593        provider
594            .storage()
595            .delete_key_package(&key_package_bundle.key_package.hash_ref(provider.crypto())?)
596            .map_err(WelcomeError::StorageError)?;
597    } else {
598        log::debug!("Key package has last resort extension, not deleting");
599    }
600    Ok((resumption_psk_store, key_package_bundle))
601}
